00554         FREE(root);
00555         return (char *)ERROR_PTR("invalid argv0; too small", procName, NULL);
00556     }
00557 
00558 #ifndef _WIN32

00561         l_int32  loc;

00563             newroot = stringNew(root + loc + 1);  /* strip out "lt-" */
00564             FREE(root);

00567         }
00568     }
00569 #else
00570     if (strstr(root, ".exe") != NULL)
00571         len -= 4;
00572 #endif  /* ! _WIN32 */
