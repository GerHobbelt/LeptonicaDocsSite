00054 /*!
00055  *  l_getDataBit()
00056  *
00057  *      Input:  line  (ptr to beginning of data line)
00058  *              n     (pixel index)
00059  *      Return: val of the nth (1-bit) pixel.
00060  */
00061 l_int32
00062 l_getDataBit(void    *line,
00063              l_int32  n)
00064 {
00065     return (*((l_uint32 *)line + (n >> 5)) >> (31 - (n & 31))) & 1;
00066 }

00103 /*!
00104  *  l_setDataBitVal()
00105  *
00106  *      Input:  line  (ptr to beginning of data line)
00107  *              n     (pixel index)
00108  *              val   (val to be inserted: 0 or 1)
00109  *      Return: void
00110  *
00111  *  Notes:
00112  *      (1) This is an accessor for a 1 bpp pix.
00113  *      (2) It is actually a little slower than using:
00114  *            if (val == 0)
00115  *                l_ClearDataBit(line, n);
00116  *            else
00117  *                l_SetDataBit(line, n);
00118  */
00119 void
00120 l_setDataBitVal(void    *line,
00121                 l_int32  n,
00122                 l_int32  val)
00123 {
00124 l_uint32    *pword;
00125 
00126     pword = (l_uint32 *)line + (n >> 5);
00127     *pword &= ~(0x80000000 >> (n & 31));  /* clear */
00128     *pword |= val << (31 - (n & 31));   /* set */
00129     return;
00130 }
